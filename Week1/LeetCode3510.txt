class Solution {
public:
    int minimumPairRemoval(vector<int>& nums) {
        const int n = nums.size();
        int ans = 0;
        int inversionsCount = 0;
        vector<int> nextIndices(n), prevIndices(n);
        vector<long> values(nums.begin(), nums.end());

        // Comparator: sort by (sum, leftmost index)
        auto comp = [](const pair<long, int>& a, const pair<long, int>& b) {
            return a.first < b.first || (a.first == b.first && a.second < b.second);
        };
        set<pair<long, int>, decltype(comp)> pairSums(comp);

        // Init linked list
        for (int i = 0; i < n; ++i) {
            nextIndices[i] = i + 1;
            prevIndices[i] = i - 1;
        }

        // Init pair sums and inversion count
        for (int i = 0; i < n - 1; ++i)
            pairSums.insert({(long)nums[i] + nums[i + 1], i});
        for (int i = 0; i < n - 1; ++i)
            if (nums[i + 1] < nums[i])
                ++inversionsCount;

        while (inversionsCount > 0) {
            ++ans;
            auto [pairSum, currIndex] = *pairSums.begin();
            pairSums.erase(pairSums.begin());

            const int nextIndex = nextIndices[currIndex];
            const int prevIndex = prevIndices[currIndex];

            // Update left neighbor
            if (prevIndex >= 0) {
                long oldSum = values[prevIndex] + values[currIndex];
                long newSum = values[prevIndex] + pairSum;
                pairSums.erase({oldSum, prevIndex});
                pairSums.insert({newSum, prevIndex});
                if (values[prevIndex] > values[currIndex]) --inversionsCount;
                if (values[prevIndex] > pairSum)           ++inversionsCount;
            }

            // Remove inversion between curr and next (being merged)
            if (values[nextIndex] < values[currIndex]) --inversionsCount;

            // Update right neighbor
            const int nextNextIndex = (nextIndex < n) ? nextIndices[nextIndex] : n;
            if (nextNextIndex < n) {
                long oldSum = values[nextIndex] + values[nextNextIndex];
                long newSum = pairSum + values[nextNextIndex];
                pairSums.erase({oldSum, nextIndex});
                pairSums.insert({newSum, currIndex});
                if (values[nextNextIndex] < values[nextIndex]) --inversionsCount;
                if (values[nextNextIndex] < pairSum)           ++inversionsCount;
                prevIndices[nextNextIndex] = currIndex;
            }

            // Merge: currIndex absorbs nextIndex
            nextIndices[currIndex] = nextNextIndex;
            values[currIndex] = pairSum;
        }

        return ans;
    }
};
