#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    string makeLargestSpecial(string s) {
        if (s.empty()) return "";
        
        vector<string> subs;
        int count = 0;
        int i = 0;
        
        // 1. 拆解頂層特殊子序列
        for (int j = 0; j < s.size(); ++j) {
            if (s[j] == '1') count++;
            else count--;
            
            if (count == 0) {
                // 2. & 3. 遞迴處理中間部分，並重新包裹
                // s.substr(i+1, j-i-1) 是去掉首尾的部分
                subs.push_back("1" + makeLargestSpecial(s.substr(i + 1, j - i - 1)) + "0");
                i = j + 1;
            }
        }
        
        // 4. 字典序降序排序
        sort(subs.begin(), subs.end(), greater<string>());
        
        // 5. 拼接
        string res = "";
        for (const string& sub : subs) res += sub;
        return res;
    }
};
