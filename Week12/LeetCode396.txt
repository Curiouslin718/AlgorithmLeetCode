class Solution {
public:
    int maxRotateFunction(vector<int>& nums) {
        int sum = 0;
        int max=0;
        for(int i=0;i<nums.size();i++){
            sum+=nums[i];
            max+=nums[i]*i;
        }

        int rotate = max;
        for(int j=nums.size()-1;j>0;j--){
            rotate = rotate - nums[j]*(nums.size()-1) + (sum-nums[j]);
            if(rotate>max){
                max = rotate;
            }
        }
        return max;
    }
};
